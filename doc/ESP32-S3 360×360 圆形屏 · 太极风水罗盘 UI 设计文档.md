1. 项目概述
本项目针对 ESP32-S3 + 360×360 纯圆形屏幕 定制一套太极风水罗盘UI系统。界面采用同心圆分层结构、东方阴阳平衡设计理念、黑金极简国风配色，结合 LVGL8.x 实现动态姿态可视化、五档状态变色、环形进度指示、极简罗盘方位标识、平滑动画与圆形触控裁切。
整体设计去繁就简、以形表意、以色分级，兼具传统罗盘文化底蕴与现代嵌入式设备的高清精致UI表现。

12.3 屏幕刷新回调与LVGL初始化
// LVGL屏幕刷新回调函数
void disp_flush_cb(lv_disp_drv_t *disp, const lv_area_t *area, lv_color_t *color_p)
{
    uint32_t width  = area->x2 - area->x1 + 1;
    uint32_t height = area->y2 - area->y1 + 1;

    tft.startWrite();
    tft.setAddrWindow(area->x1, area->y1, area->x2, area->y2);
    tft.pushColors((uint16_t *)&color_p->full, width * height, true);
    tft.endWrite();

    lv_disp_flush_ready(disp);
}

// LVGL完整初始化（含圆形屏幕裁剪）
void lvgl_system_init(void)
{
    // 屏幕硬件初始化
    tft.begin();
    tft.setRotation(0);
    tft.fillScreen(TFT_BLACK);

    // LVGL内核初始化
    lv_init();
    lv_disp_draw_buf_init(&draw_buf, draw_buffer, NULL, SCREEN_W * 80);

    // 注册显示设备驱动
    static lv_disp_drv_t disp_drv;
    lv_disp_drv_init(&disp_drv);
    disp_drv.hor_res  = SCREEN_W;
    disp_drv.ver_res  = SCREEN_H;
    disp_drv.flush_cb = disp_flush_cb;
    disp_drv.draw_buf = &draw_buf;

    // 开启核心功能：圆形屏幕裁剪（屏蔽四角无效区域）
    lv_disp_set_round(&disp_drv, true);
    lv_disp_set_center(&disp_drv, CENTER_X, CENTER_Y);
    lv_disp_set_radius(&disp_drv, VALID_RADIUS);

    lv_disp_drv_register(&disp_drv);
}

12.6 姿态状态分级更新核心逻辑
// 根据倾角更新UI状态色与文本
void update_compass_state(float tilt_angle)
{
    lv_color_t state_color;
    char state_buf[32];

    // 五档状态分级判断（匹配文档规范）
    if(fabs(tilt_angle) < 2.0f)
    {
        state_color = COLOR_STATE_NORMAL;
        sprintf(state_buf, "Balance OK");
    }
    else if(fabs(tilt_angle) < 8.0f)
    {
        state_color = COLOR_STATE_LIGHT;
        sprintf(state_buf, "Slight Tilt");
    }
    else if(fabs(tilt_angle) < 15.0f)
    {
        state_color = COLOR_STATE_MID;
        sprintf(state_buf, "Medium Tilt");
    }
    else if(fabs(tilt_angle) < 25.0f)
    {
        state_color = COLOR_STATE_HEAVY;
        sprintf(state_buf, "Heavy Tilt");
    }
    else
    {
        state_color = COLOR_STATE_DANGER;
        sprintf(state_buf, "Danger Tilt");
    }

    // 更新进度环颜色
    lv_obj_set_style_arc_color(ui_state_arc, state_color, LV_PART_INDICATOR);
    // 更新状态文本
    lv_label_set_text(ui_main_text, state_buf);
    // 更新倾角数值
    sprintf(state_buf, "%.2f°", tilt_angle);
    lv_label_set_text(ui_sub_text, state_buf);

    // 进度环角度映射
    uint16_t progress = (uint16_t)constrain(fabs(tilt_angle) * 4, 0, 360);
    lv_arc_set_end_angle(ui_state_arc, progress);
}
